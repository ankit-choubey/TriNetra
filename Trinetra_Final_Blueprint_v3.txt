2.  MATHEMATICAL ENGINE  (COMPLETE + FORMAL)
This section defines ALL formulas with normalization bounds, outlier handling, and sentiment computation. This is the answer to 'walk me through your logic'.
2.1  Feature Normalization Strategy
All features must be normalized to [0, 1] before weighted summation. Raw values are stored separately in derived_features. Normalized values are stored in derived_features.*_normalized.
# Normalization bounds (define in risk-agent/normalization.py) FEATURE_BOUNDS = {     # (raw_min, raw_max, risk_direction)     # risk_direction: 'inverse' = lower raw = higher risk     #                 'direct'  = higher raw = higher risk     "dscr":             (0.5,  2.5,  "inverse"),   # DSCR < 0.5 = max risk     "icr":              (1.0,  5.0,  "inverse"),     "leverage":         (0.0,  5.0,  "direct"),    # D/E > 5 = max risk     "ccc":              (0,    180,  "direct"),     # CCC days > 180 = max risk     "revenue_growth":   (-0.3, 0.4,  "inverse"),   # growth < -30% = max risk     "ebitda_margin":    (-0.1, 0.4,  "inverse"),     "gst_discrepancy":  (0.0,  0.5,  "direct"),    # discrepancy > 50% = max risk     "circular_trade":   (0.0,  0.5,  "direct"),     "litigation_count": (0,    10,   "direct"),     "news_sentiment":   (-1.0, 1.0,  "inverse"),   # -1 = very negative = high risk }  def normalize(raw_value, feat_name):     lo, hi, direction = FEATURE_BOUNDS[feat_name]     # Clip to bounds     clipped = max(lo, min(hi, raw_value))     ratio   = (clipped - lo) / (hi - lo)     # Convert to risk score: inverse = 1-ratio gives risk     return (1 - ratio) if direction == "inverse" else ratio
2.2  Weighted Risk Score
# Risk weights (must sum to 1.0) WEIGHTS = {     "dscr_normalized":          0.25,     "leverage_normalized":      0.18,     "revenue_growth_normalized":0.12,     "ebitda_margin_normalized": 0.10,     "gst_discrepancy_norm":     0.12,     "circular_trade_norm":      0.08,     "litigation_norm":          0.10,     "news_sentiment_norm":      0.05, } # Total = 1.00  def compute_risk_score(features: dict) -> float:     score = sum(WEIGHTS[k] * features[k] for k in WEIGHTS if k in features)     # Apply PD adjustment AFTER weighted sum     pd_adj = features.get("pd_risk_adjustment", 0.0)     return round(min(1.0, max(0.0, score + pd_adj)), 4)  # Band assignment def assign_band(score: float) -> str:     if score < 0.30: return "LOW"     if score < 0.55: return "MEDIUM"     if score < 0.75: return "HIGH"     return "REJECT"  # Limit and rate def compute_limit_and_rate(requested: float, score: float, base_rate_bps: float) -> tuple:     limit     = requested * (1 - score * 0.6)    # max 60% haircut     rate_bps  = base_rate_bps + (score * 400)    # max +400bps premium     return round(limit, 2), round(rate_bps, 2)

# decision_confidence/calculator.py  def compute_confidence(ucso: dict) -> dict:     docs     = ucso.get("documents", {}).get("files", [])     CRITICAL = ["ANNUAL_REPORT","BANK_STMT","GST_RETURN","ITR"]     present  = [d for d in docs if d["type"] in CRITICAL and d["parsed"]]     data_completeness = len(present) / len(CRITICAL)  # 0–1      confidences = [d["confidence"] for d in docs if d["parsed"] and d["confidence"] > 0]     extraction_confidence = sum(confidences)/len(confidences) if confidences else 0.0      kb_hours = ucso.get("web_intel",{}).get("kb_freshness_hours", 9999)     MAX_HOURS = 168  # 1 week     kb_freshness_score = max(0.0, 1 - (kb_hours / MAX_HOURS))      # model_stability: 1.0 if no drift detected, decrements on each drift event     model_stability_score = 1.0 if not ucso.get("risk",{}).get("drift_detected") else 0.5      # Weighted formula     score = (0.30 * data_completeness            + 0.30 * extraction_confidence            + 0.20 * kb_freshness_score            + 0.20 * model_stability_score)      return {         "score":                  round(score, 4),         "data_completeness":      round(data_completeness, 4),         "extraction_confidence":  round(extraction_confidence, 4),         "kb_freshness_score":     round(kb_freshness_score, 4),         "model_stability_score":  round(model_stability_score, 4),         "formula": "0.30×DC + 0.30×EC + 0.20×KF + 0.20×MS"     }

# web-agent/sentiment.py # pip install vaderSentiment from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer  analyzer = SentimentIntensityAnalyzer()  def score_article(headline: str, body: str = "") -> dict:     text     = f"{headline}. {body}"     scores   = analyzer.polarity_scores(text)     compound = scores["compound"]          # -1.0 to +1.0     # Normalize to risk contribution (0=no risk, 1=max risk)     risk_contribution = (1 - compound) / 2  # maps [-1,1] → [1,0]     return {         "sentiment_score":    round(compound, 4),         "risk_contribution":  round(risk_contribution, 4),         "label": "POSITIVE" if compound > 0.05 else "NEGATIVE" if compound < -0.05 else "NEUTRAL"     }  def aggregate_news_sentiment(articles: list) -> float:     if not articles: return 0.5   # neutral default     return sum(a["risk_contribution"] for a in articles) / len(articles)

# bank-recon-agent/reconciliation.py  def reconcile_bank_vs_gst(bank_statements: list, gst_returns: list, itr_income: float) -> dict:     # Step 1: Sum all bank credit entries (exclude inter-account transfers)     EXCLUDE_KEYWORDS = ["TRF","NEFT TRF","RTGS TRF","IMPS TRF","OWN ACCOUNT"]     bank_credit_total = sum(         entry["credit_amount"] for stmt in bank_statements         for entry in stmt["transactions"]         if entry["type"] == "CREDIT"         and not any(kw in entry.get("narration","").upper() for kw in EXCLUDE_KEYWORDS)     )      # Step 2: Sum GST reported turnover (from GSTR-1 or 3B)     gst_turnover = sum(r.get("total_taxable_turnover", 0) for r in gst_returns)      # Step 3: Divergence calculation     if gst_turnover > 0:         divergence = abs(bank_credit_total - gst_turnover) / gst_turnover     else:         divergence = 1.0   # max mismatch if GST = 0      # Step 4: Revenue inflation flag     inflation_flag = bank_credit_total > gst_turnover * 1.3      # Step 5: Detect round-trip transactions (same amount in+out within 3 days)     round_trips = detect_round_trips(bank_statements, window_days=3)      # Step 6: Bounce count     bounces = sum(1 for stmt in bank_statements                   for t in stmt["transactions"]                   if "BOUNCE" in t.get("narration","").upper()                   or "RETURN" in t.get("narration","").upper())      verdict = "OK"     if divergence > 0.30 or inflation_flag: verdict = "INFLATION_SUSPECTED"     elif divergence > 0.15: verdict = "MISMATCH"      return {         "bank_credit_turnover":  round(bank_credit_total, 2),         "gst_reported_turnover": round(gst_turnover, 2),         "itr_reported_income":   itr_income,         "turnover_divergence_pct": round(divergence * 100, 2),         "revenue_inflation_flag": inflation_flag,         "round_trip_transactions": round_trips,         "bounce_count_last_12m": bounces,         "reconciliation_verdict": verdict     }
